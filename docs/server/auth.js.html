/**
 * This file contains code to enable OAuth functionality on the backend.
 * @author Krishnan Prashanth
 * @author Neha Sujith
 */
const router = require("express").Router();
const passport = require("passport");

/**
 * Authenticate login and display the appropriate success or failure message
 * @param {*} req the request parameters
 * @param {*} res the response body 
 */
router.get("/login/success", (req, res) => {
	if (req.user) {
		res.status(200).json({
			error: false,
			message: "Successfully Logged In",
			user: req.user,
		});
	} else {
		res.status(403).json({ error: true, message: "Not Authorized" });
	}
});

/**
 * Display login failure message
 * @param {*} req the request paramters
 * @param {*} res the response body
 */
router.get("/login/failed", (req, res) => {
	res.status(401).json({
		error: true,
		message: "Log in failure",
	});
});

router.get("/google", passport.authenticate("google", ["profile", "email"]));

router.get(
	"/google/callback",
	passport.authenticate("google", {
		successRedirect: process.env.CLIENT_URL,
		failureRedirect: "/login/failed",
	})
);

/**
 * Logout the user
 * @param {*} req the request parameters
 * @param {*} the response body
 */
router.get("/logout", (req, res) => {
	req.logout();
	res.redirect(process.env.CLIENT_URL);
});

module.exports = router;
