/**
 * This file contains the logic to access the store's financial data from the database.
 * @author Krishnan Prashanth
 */
const express = require("express");
const app = express.Router();
const db = require("../db");

/**
 * Gets all financial data about the store from the database. All attributes are retrieved.
 * @param {*} req the request parameters
 * @param {*} res the response body
 */
app.get("/financeItems", async (req, res) => {
  try {
    const results = await db.query("SELECT * FROM finances;");

    res.status(200).json({
      status: "success",
      results: results.rows.length,
      data: {
        table: results.rows,
      },
    });
  } catch (err) {
    console.log(err);
  }
});


module.exports = app;
